109        Metodo que recupera los datos almacenados en cache.

114            str: Datos del archivo

 17        """
 18        Convierte la url en un string de tipo hash para facilitar su acceso posterior
 19
 20        Args:
 21            url (str): url de internet
 22        Returns:
 23            str: Hash MD5 en hexadecimal de 32 caracteres

 28    def get(self, url: str, **kwargs) -> str:
 29        """
 30        Descarga el contenido especificado de una URL de internet
 31
 32        Args:
 33            url (str): url de internet
 34        Returns:
 35            str: el contenido de la url especificada.
 36        Raises:
 37            CacheError: Si el status code de response es diferente a 200
 38        """
 39        url_hash = self.__url_to_hash(url)
 40        if self.exists(url):
 41            return self.load(url)
 42
 43        # Descarga de internet
 44        try:
 45            response = requests.get(url, timeout=30)
 46            if response.status_code == 200:
 47                self.set(url_hash, response.text)
 48                return response.text
 49            else:
 50                raise CacheError(f'Error HTTP {response.status_code} to download {url}')
 51        except requests.exceptions.RequestException as e:
 52            raise CacheError(f'Error {e}')
